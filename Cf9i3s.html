문제

A양과 B군이 가위바위보를 선택하면 승패를 출력하는 문제입니다 실행순서를 클릭하세요 
A양, 가위, 바위, 보 중 하나를 선택하세요: 바위
바위를 냈습니다.
B군, 가위, 바위, 보 중 하나를 선택하세요: 가위
가위를 냈습니다.
A양이 이겼습니다.

프로그램 코드

	#include <stdio.h>

	#include <string.h>


1	int main() {

		char playerA[10] = "";

2		char playerB[10] = "";


		printf("A양, 가위, 바위, 보 중 하나를 선택하세요: ");

		scanf( "%s", playerA );

3		printf( "%s를 냈습니다.\n", playerA );


		printf("B군, 가위, 바위, 보 중 하나를 선택하세요: ");

		scanf( "%s", playerB );

4		printf( "%s를 냈습니다.\n", playerB );


5		if ( !strcmp( playerA, playerB ) ) {

6			printf( "\nA양과 B군이 비겼습니다.\n" );

		}

7		else if ( !strcmp( playerA, "가위" ) && !strcmp( playerB, "보" ) )	 {

8			printf( "\nA양이 이겼습니다.\n" );

		}

9		else if ( !strcmp( playerA, "바위" ) && !strcmp( playerB, "가위" ) ) {

10			printf( "\nA양이 이겼습니다.\n" );

		}

11		else if ( !strcmp( playerA, "보" ) && !strcmp( playerB, "바위" ) ) {

12			printf( "\nA양이 이겼습니다.\n" );

		}

13		else {

14			printf( "\nB군이 이겼습니다.\n" );

		}


15		return 0;

	}
