알고리즘

프로그램시작 
   단어들 입력 
   단어 정렬 
      마지막 칸에 가장 큰 값을 저장하면서 한 칸씩 앞으로 영역 좁혀가기 
         영역의 첫번째 칸부터 인접한 두 값을 비교하면서 
         앞칸 단어가 뒤칸 단어보다 알파벳순으로 뒤면, 두 단어를 맞바꾸기 
   정렬된 단어 출력 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter09/WordSort.c

 	#include <stdio.h>

 	#include <string.h>

	 

	// 프로그램시작 

1	int main() {

		char word[3][32] = { "", "", "" };

		char temp[32] = "";

		int index = 0;

2		int last = 0;

 

		 

		// 단어들 입력  

3		for( index = 0; index < 3; index++ )

		{

			printf( "단어를 입력하세요: " );

4			scanf( "%s", word[index] );

		}

 

		// 단어 정렬 

		// 마지막 칸에 가장 큰 값을 저장하면서 한 칸씩 앞으로 영역 좁혀가기 

5		for ( last = 2; last >= 0; last-- )

		{

			// 영역의 첫번째 칸부터 인접한 두 값을 비교하면서  

6			for ( index = 0; index < last; index++ )

			{

				// 앞칸 단어가 뒤칸 단어보다 알파벳순으로 뒤면, 두 단어를 맞바꾸기 

7				if ( strcmp( word[index], word[index+1] ) > 0 )

				{

8					strcpy( temp, word[index] );

9					strcpy( word[index], word[index+1] );

10					strcpy( word[index+1], temp );

				}

			}

		}

	 

		// 정렬된 단어 출력 

11		for ( index = 0; index < 3; index++ )

		{

12			printf( "%d) %s ", index + 1, word[index] );

		}

	 

		// 프로그램 종료 

13		return 0;

	}
