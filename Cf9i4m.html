문제

사용자가 입력한 단어를 알파벳순으로 정렬하는 문제입니다 이를 해결하는 다음 프로그램의 실행상태를 확인하세요 
단어를 입력하세요: 사과
단어를 입력하세요: 바나나
단어를 입력하세요: 배
단어를 입력하세요: 포도
단어를 입력하세요: 귤
1) 귤 2) 바나나 3) 배 4) 사과 5) 포도

프로그램 코드

	 	#include <stdio.h>
 	#include <string.h>
	
1	int main() {																														
		char word[3][32] = { "", "", "" };
		char temp[32] = "";
		int index = 0;	
2		int last = 0;
		
3		for( index = 0; index < 3; index++ )
		{																																			
			printf( "단어를 입력하세요: " );
4			scanf( "%s", word[index] );
		} 																																		
5		for ( last = 2; last >= 0; last-- )
		{																																			
6			for ( index = 0; index < last; index++ )
			{																																		
7				if ( strcmp( word[index], word[index+1] ) > 0 ) 
				{																																	
8					strcpy( temp, word[index] );					
9					strcpy( word[index], word[index+1] );				
10					strcpy( word[index+1], temp );						
				}																																	
			}																																		
		}																																			
	
11		for ( index = 0; index < 3; index++ )
		{																																			
12			printf( "%d) %s ", index + 1, word[index] );
		} 																																		
	
13		return 0;
	}
